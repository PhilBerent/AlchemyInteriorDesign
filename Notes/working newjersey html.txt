    <div class="arrow arrow-right" onclick="nextImage()">&#8250;</div>

    <script>
        let currentImage = 2;
        const totalImages = 6; // Number of images for New Jersey Country House (images 2 to 6)

        function updateImage() {
            // Adjust to match the correct lowercase path and .jpeg extension
            document.getElementById('project-image').src = `../images/newjersey/page_${currentImage}.jpeg`;
        }

        function prevImage() {
            if (currentImage > 2) {
                currentImage--;
                updateImage();
            }
        }

        function nextImage() {
            if (currentImage < totalImages) {
                currentImage++;
                updateImage();
            }
        }
    </script>
